      function displayFootprint(f, m , c, p) {
        let message = "Your Footprint: " + f +" --  This is a "+c+" Footprint.\n" +m +"\n"+p
        document.getElementById('output-text').value = (message);
      }

      function validateYesNoInput(inputId) {
        let inputElement = document.getElementById(inputId);
        let pattern = /^(yes|no)$/i;

        inputElement.addEventListener('input', function() {
          let isValid = pattern.test(this.value.toLowerCase());

          if (!isValid) {
            this.setCustomValidity("Please enter 'yes' or 'no'");
          } else {
            this.setCustomValidity('');
          }
        });
      }

      window.onload = function() {
        validateYesNoInput('newspaper-factor');
        validateYesNoInput('tin-factor');
      };

      function calculateFootprint() {
        // Validation logic
        let newspaperEmissionInput = document.getElementById('newspaper-factor');
        let tinEmissionInput = document.getElementById('tin-factor');

        if (!newspaperEmissionInput.checkValidity() || !tinEmissionInput.checkValidity()) {
          // If the validation fails, display an error message or take appropriate action
          alert('Please enter "yes" or "no" for newspaper and tin recycling.');
          return;
        }

        // If validation passes, proceed with the calculation
        let electricEmission = document.getElementById('electricity-factor').value;
        let gasEmission = document.getElementById('gas-factor').value;
        let oilEmission = document.getElementById('oil-factor').value;
        let carEmission = document.getElementById('carmileage-factor').value;
        let longhaulEmission = document.getElementById('longhaul-factor').value;
        let shorthaulEmission = document.getElementById('shorthaul-factor').value;
        let newspaperEmission = newspaperEmissionInput.value.toLowerCase();
        let tinEmission = tinEmissionInput.value.toLowerCase();

        // XMLHttpRequest logic
        let xhttp = new XMLHttpRequest();
        xhttp.onreadystatechange = function() {
          if (this.readyState == 4 && this.status == 200) {
            var j = JSON.parse(this.response);
            let footprint = j.footprint;
            let category = j.category;
            let proceedMsg = j.trackPrompt;
            let message = j.message;

            displayFootprint(footprint, message, category, proceedMsg);
          }
        };
        xhttp.open("POST", "/calculatefootprint", true);
        xhttp.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
        xhttp.send("electricity-factor=" + electricEmission +
                   "&gas-factor=" + gasEmission +
                   "&oil-factor=" + oilEmission +
                   "&carmileage-factor=" + carEmission +
                   "&longhaul-factor=" + longhaulEmission +
                   "&shorthaul-factor=" + shorthaulEmission +
                   "&newspaper-factor=" + newspaperEmission +
                   "&tin-factor=" + tinEmission);

        return false;
      }
    </script>
